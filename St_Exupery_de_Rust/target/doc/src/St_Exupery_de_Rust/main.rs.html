26/// main() contient le bloc d'exécution principal du programme qui orchestre la progression dans
27/// le jeu.
28/// 
29/// Declare les entités principales et appelle main_menu() pour les initialiser 
30/// selon le choix de difficulté de l'utilisateur.
31/// 
32/// main() gère la boucle principale de navigation du monde de jeu et permet
33/// de progresser vers l'objectif en faisant appel à world_navigate, world_get_scenario
34/// et world_set_scenario selon l'évaluation de 'match' pour un enum 'Nav'.
35/// 
36/// Permet de quitter le programme lorsque le joueur choisit de quitter à partir du menu principal ou
37/// le menu de navigation.
38/// 
39/// #Arguments
40/// 
41/// * Aucuns
42/// 
43/// #Retourne
44/// 
45/// * ( )
46/// 
47
48fn main() {
49
50    let (mut world_entity_hero, mut world_entity_monsters) = main_menu();
51
52    if world_entity_monsters.len()==0 {
53        println!("\nFermeture...\n");
54        process::exit(0);
55    }
56
57    let mut world_hero_death =false;
58
59    loop {
60        let world_nav_choice = world_navigate(&world_entity_monsters, &mut world_entity_hero);
61
62        match world_nav_choice {
63            Nav::Explorer => {
64
65                let world_scenario = world_get_scenario();
66
67                world_hero_death=world_set_scenario(world_scenario, &mut world_entity_monsters, &mut world_entity_hero);
68
69                if world_hero_death == true {
70                    (world_entity_hero, world_entity_monsters)=main_menu();
71
72                    if world_entity_monsters.len()==0 {
73                        println!("\nFermeture...\n");
74                        process::exit(0);
75                    }
76                }
77
78
79            }
80            Nav::Guerir => {
81                world_entity_hero.boire_potion();
82            }
83            Nav::Quit => {
84                println!("\nQuitting...");
85                process::exit(0);
86            }
87        }
88
89        if world_entity_monsters.len() ==0 {
90
91            thread::sleep(Duration::from_millis(2000));
92            println!("\n{}", "Toutes les créatures de ce territoire sont anéanties...".truecolor(230, 112, 11));
93            thread::sleep(Duration::from_millis(2000));
94            println!("\n{}","Vous prenez un genou et déposez votre arme.  Vous jurez en silence que Rust verra l'aube...".truecolor(230, 112, 11));
95            thread::sleep(Duration::from_millis(5000));
96
97            println!("\n");
98
99            (world_entity_hero, world_entity_monsters) = main_menu();
100            
101            if world_entity_monsters.len()==0 {
102                println!("\nFermeture...\n");
103                process::exit(0);
104            }
105            
106        }
107
108    }
109
110}

112/// Présente le menu principal au joueur et recoit son choix de difficulté du jeu.
113/// Appelle la fonction world_initialize_entities en fonction du choix.
114/// 
115/// #Arguments
116/// 
117/// * Aucuns
118/// 
119/// #Retourne
120/// 
121/// * (StExupery, Vec<Monstre>)
122/// 
123/// # Exemples
124/// 
125/// '''
126/// let (mut hero, mut monstres) = main_menu();
127/// '''
128fn main_menu() -> (StExupery,Vec<Monstre>) {
129    
130    let mut main_menu_sel = String::new();
131
132    let mut hero = StExupery::default();
133    let mut world_monsters = Vec::<Monstre>::default();
134
135    println!("{text:^width$}", text="St-Exupéry de Rust".on_truecolor(212,151,11), width=50);
136    thread::sleep(Duration::from_secs(1));
137
138
139    println!("\nChoisir le niveau de défi:\n\n (F)acile,\n (M)oyen\n (D)ifficile\n (Q)uitter\n");
140    main_menu_sel = get_validated_menu_input(&vec!["f".to_string(),
141                                                        "m".to_string(),
142                                                        "d".to_string(),
143                                                        "q".to_string()]);
144
145    if main_menu_sel =="q" {
146        return (hero,world_monsters)
147    } else {
148        hero = world_initialize_entities(Diff::from_str(&main_menu_sel.to_lowercase()).unwrap(), 
149                                                        &mut world_monsters);
150    }
151
152    (hero, world_monsters)
153
154}

156/// Utilise io::stdin pour recevoir la saisie de l'utilisateur au terminal et 
157/// la retourner de la fonction.
158/// 
159/// #Arguments
160/// 
161/// * Aucuns
162/// 
163/// #Retourne
164/// 
165/// * String
166/// 
167/// # Exemples
168/// 
169/// '''
170/// let choice = get_input();
171/// '''
172fn get_input() -> String {
173    let mut input_string = String::new();
174    print!("+∩+ >> ");
175    let _ = io::stdout().flush();
176    io::stdin()
177            .read_line(&mut input_string)
178            .expect("Failed to read line.");
179
180        input_string.trim().to_string()
181}

183/// Recoit un vecteur de valeurs valides pour la saisie au terminal de l'utilisateur
184/// et appelle get_input().  Si la saisie n'est pas valide demande de saisir à nouveau.
185/// 
186/// #Arguments
187/// 
188/// * 'valid'
189/// 
190/// #Retourne
191/// 
192/// * String
193/// 
194/// # Exemples
195/// 
196/// '''
197/// let valid_choices = ["a", "b", "c"];
198/// 
199/// let validated_menu_option = get_validated_menu_input(valid_choices);
200/// 
201/// '''
202fn get_validated_menu_input(valid: &Vec<String>) -> String {
203    
204    let mut input = String::new();
205    loop{
206        input.clear();
207        input = get_input();
208        
209        if valid.contains(&input.to_lowercase()) {
210            return input
211        } else {
212            println!("Votre choix est invalide.")
213        }
214    }
215}

217/// 'Match' la valeur de 'difficulty'(enum Diff) et initialise les modèles Monstre et StExupery
218/// en fonction.
219/// 
220/// #Arguments
221/// 
222/// * 'difficulty', 'enemies'
223/// 
224/// #Retourne
225/// 
226/// * StExupery
227/// 
228/// # Exemples
229/// 
230/// '''
231/// let mut monsters = Vec::<Monstre>::new();
232/// let mut hero = world_initialize_entities(Diff::F, &mut monsters);
233/// 
234/// '''
235fn world_initialize_entities (difficulty: Diff, enemies: &mut Vec<Monstre>) -> StExupery {
236    
237    match difficulty {
238        Diff::F => {
239            for _i in 0..5 {
240                enemies.push(Monstre::new(&difficulty));
241            }
242
243            return StExupery::new(&difficulty);
244        }
245
246        Diff::M => {
247            for _i in 0..7 {
248                enemies.push(Monstre::new(&difficulty));
249            }
250
251            return StExupery::new(&difficulty);
252        }
253
254        Diff::D => {
255            for _i in 0..10 {
256                enemies.push(Monstre::new(&difficulty));
257            }
258
259            return StExupery::new(&difficulty);
260        }
261    }
262
263}

335/// Lorsque le joueur choisit d'explorer le monde du jeu, la fonction choisit un de quatres 
336/// scénarios exploratoires possibles dans le jeu (combat, ou trouver un bouclier, une épée ou une potion) 
337/// de manière aléatoire.
338/// 
339/// #Arguments
340/// 
341/// * Aucuns
342/// 
343/// #Retourne
344/// 
345/// * Enum 'Scenario'
346/// 
347/// # Exemples
348/// 
349/// '''
350/// let scene = world_get_scenario();
351/// '''
352fn world_get_scenario() -> Scenario {
353
354    let rng = rand::rng;
355    let select: i16 = rng().random_range(1..9);
356
357    if select == 1 {
358        Scenario::Potion
359    } else if select == 2 {
360        Scenario::Epee((Loot::default().find_sword()).clone())
361    } else if select == 3 {
362        Scenario::Bouc(Loot::default().find_shield().clone())
363    } else {
364        Scenario::Monstre
365    }
366
367}
